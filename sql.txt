-- DRAFT – pattern from "Text 1/2/3 etc. parameters show OTHER in Data Lab" article,
-- adapted for text13 and BA campaign 26690.
SELECT text13, COUNT(*)
FROM action_fact
WHERE action_datetime >= '2025-01-01'
  AND campaign_id = 26690
GROUP BY 1
ORDER BY 2 DESC;


-- DRAFT – same structure as the vetted text-param allowlist query,
-- restricted to text_num = 13 for this campaign.
SELECT *
FROM text_event_param_allowed_mappings
WHERE campaign_id = 26690
  AND text_num = 13;
