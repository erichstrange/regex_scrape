/* DRAFT – Prospects created for MobileX (25438) Oct 15–31, 2025 */
SELECT
  BIN_TO_UUID(r.rsi, 1)                        AS relationship_id,
  BIN_TO_UUID(r.supply_psi, 1)                 AS brand_supply_psi,
  BIN_TO_UUID(r.demand_psi, 1)                 AS partner_demand_psi,
  rst.name                                     AS source_type,          -- precise source label behind UI's "Other"
  rstat.name                                   AS relationship_state,   -- e.g., PROSPECT/OPEN/CLOSED
  r.doe                                        AS created_utc,
  r.uoe                                        AS created_by_username,  -- who inserted (user/service)
  r.ulu                                        AS last_updated_by_username
  /* Optional: pull a partner name if available */
, COALESCE(c.name, p.name)                      AS partner_name
FROM relationship r
JOIN relationship_source_type rst
  ON rst.id = r.relationship_source_type_id
JOIN relationship_state_type rstat
  ON rstat.id = r.relationship_state_type_id
/* Anchor to MobileX’s program via program_platform → campaign id (no UUID juggling needed) */
JOIN program_platform pp
  ON pp.psi = r.supply_psi
  AND (
        pp.legacy_id = 25438         /* use this */
        /* OR pp.ircm_campaign_id = 25438  -- uncomment if A in step 1 failed and B worked */
      )
/* Partner name (best‑effort: company or program) */
LEFT JOIN company  c ON c.psi = r.demand_psi
LEFT JOIN program  p ON p.psi = r.demand_psi
WHERE r.doe >= '2025-10-15' AND r.doe < '2025-11-01'
ORDER BY r.doe DESC;
