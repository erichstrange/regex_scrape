SELECT action_date,
       oid,
       campaign_id,
       JSON_EXTRACT_STRING(metadata,'conv_ref_url') AS conv_ref_url
FROM ir_noreferrer_event
WHERE campaign_id = 25209
  AND action_type = 'CONV'
  AND action_status = 'FAILURE'
  AND action_date >= '2025-10-03'
ORDER BY action_date DESC
LIMIT 200;
