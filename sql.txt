SELECT fi.invoice_number,
       fi.invoice_date, fi.period_start, fi.period_end,
       fil.line_number, fil.event_type_name, fil.quantity,
       fil.unit_amount, fil.amount, fil.description,
       fil.created_by, fil.created_date, fil.updated_by, fil.updated_date
FROM finance_invoice fi
JOIN finance_invoice_line_item fil ON fil.finance_invoice_id = fi.id
WHERE fi.invoice_number = '170911-3482';
