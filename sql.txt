SELECT
  id,
  state,
  debit_amount,
  debit_amount_currency,
  credit_amount,
  credit_amount_currency,
  due_date,
  charge_date,
  link_date,
  description,
  reference,
  debit_ledger_id,
  credit_ledger_id
FROM finance_invoice
WHERE seller_finance_account_id = 2198017
  AND (debit_ledger_id = 40811510 OR credit_ledger_id = 40811510)
ORDER BY id DESC;
