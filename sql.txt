-- DRAFT: list recommendations created for MobileX program
SELECT iram_advertiser_id     AS advertiser_id,
       ircm_campaign_id       AS campaign_id,
       iram_publisher_id      AS partner_id,
       source,                -- e.g., USER / ALGORITHM / etc. (field names may differ)
       created_by_user_id,    -- link to core_estaleauser.id
       doe
FROM iop.iop_partnership_recommendation
WHERE iram_advertiser_id = 5475947
  AND ircm_campaign_id   = 25438
ORDER BY doe DESC;
