SELECT
  cf.campaign_id, cf.event_datetime, JSON_EXTRACT_STRING(cf.json,'inboundLink') AS tracking_link,
  cf.click_id, cf.profile_id, ad.ad_id, ad.name AS ad_name,
  JSON_EXTRACT_STRING(cf.json,'blockingRule') AS blocking_rule,
  JSON_EXTRACT_STRING(cf.json,'ruleName')     AS rule_name,
  cf.landing_page_url, cf.method, cf.classification
FROM click_fact cf
LEFT JOIN ad_dim ad ON cf.ad_dim_id = ad.id
WHERE cf.advertiser_id = 173567
  AND cf.publisher_id  = 376373
  AND cf.campaign_id   = 3899
  AND ad.ad_id         = 797408
  AND cf.event_datetime >= '2025-09-03 19:24:37'
  AND cf.event_datetime <= '2025-09-03 20:24:37'
ORDER BY cf.event_datetime DESC;
