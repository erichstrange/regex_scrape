SELECT
  sc.iram_publisher_id,
  sc.ircm_campaign_id,
  sc.date_publisher_signed,
  sc.date_advertiser_signed,
  sc.is_pub_pending,
  sc.is_adv_pending,
  sc.username_publisher_signer,
  sc.username_advertiser_signer
FROM ircm_signedcontract sc
WHERE sc.iram_publisher_id = '5945725'
  AND sc.ircm_campaign_id = '21972'
  AND (sc.date_contract_end IS NULL OR sc.date_contract_end = '')
;
