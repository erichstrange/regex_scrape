SELECT doe, publisher_id, ad_id, campaign_id, method, landing_page_url, ref_url, user_agent
FROM irer_click_v3
WHERE campaign_id = 5335
  AND ad_id = 1063662
  AND publisher_id IN (3019504, 2197213)
  AND doe >= NOW() - INTERVAL '3 days'
ORDER BY doe DESC
LIMIT 500;
