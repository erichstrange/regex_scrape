SELECT
    tt.template_term_id,
    tt.name AS template_term_name,
    ttem.event_type_id,
    ttem.crediting_rule,
    tt.status,
    tt.effective_date
FROM template_term_event_map ttem
JOIN template_term tt ON ttem.template_term_id = tt.template_term_id
WHERE ttem.account_id = 3256060
AND ttem.event_type_id = 29434;
