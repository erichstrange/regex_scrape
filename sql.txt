SELECT
  event_datetime,
  program_id,
  event_type_code,
  text2,
  numeric1,
  payload_json
FROM   hubspot_plugin_event_fact          -- table is in r_ds_ods
WHERE  program_id = 6123248
  AND  event_datetime >= date_trunc('day', current_date) - INTERVAL '1 day'
ORDER  BY event_datetime DESC
LIMIT 200;
