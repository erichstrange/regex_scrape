SELECT ref_ip_address,
       COUNT(*) AS action_cnt,
       MIN(action_date) AS first_seen,
       MAX(action_date) AS last_seen
FROM   irac_action2
WHERE  iram_advertiser_id = 5333971        -- Brooks Running
  AND  publisher_id       = 5133335        -- SallyGarmon
  AND  action_date BETWEEN '2025-05-01' AND '2025-06-10'
GROUP  BY ref_ip_address
HAVING COUNT(*) > 10                       -- adjust if too many / too few
ORDER  BY action_cnt DESC;
