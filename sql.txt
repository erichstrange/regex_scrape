-- VETTED: "Account Features"
SELECT a.feature_key AS "Key",
       a.name        AS "Feature Name"
FROM iram_feature AS a
JOIN iram_feature_auth_account_feature AS b
  ON a.id = b.iram_feature_id
JOIN iram_feature_auth_account AS c
  ON c.id = b.iram_feature_auth_account_id
WHERE iram_feature_auth_account_id IN (
   SELECT iram_feature_auth_account_id
   FROM iram_account AS d
   JOIN core_account AS e ON d.id = e.id
   WHERE d.id = 6737279  -- MP account id
);
