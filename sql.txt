SELECT
    ap.iram_account_id,
    ap.ircm_campaign_id,
    c.campaign_name            AS campaign_name,
    c.ircm_campaign_id         AS ircm_campaign_id
FROM
    iram_advertiser_publisher ap
JOIN
    ircm_campaign c
      ON c.ircm_campaign_id = ap.ircm_campaign_id
WHERE
    ap.iram_account_id = [[ACCOUNT_ID]]
LIMIT 50;
