-- DRAFT – Publisher country history
-- DS: r_ds_irdw

SELECT
  publisher_id,
  date_from,
  date_to,
  country,
  state
FROM publisher_dim
WHERE publisher_id = 6564108
ORDER BY date_from;

SELECT
  publisher_id,
  publisher_name,
  country,
  state,
  city,
  source_doe,
  source_dlu,
  doe,
  dlu
FROM publisher_account
