-- 1️⃣  Did TI even ingest the file?
SELECT trace_id,
       filename,
       submit_timestamp,
       status,
       total_row_count,
       error_row_count
FROM   irdw_prod.ti_file
WHERE  filename LIKE 'pvc_%'           -- or 'kevin%' / 'sophie%'
ORDER  BY submit_timestamp DESC
LIMIT  10;

-- 2️⃣  If error_row_count > 0, what broke?
--    (replace <trace_id>)
SELECT row_number,
       error_code,
       error_message
FROM   irdw_prod.ti_error
WHERE  trace_id = '<trace_id>';
