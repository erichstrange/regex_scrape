SELECT
  campaign_id,
  sku,
  version,
  date_from,
  date_to,
  current_price,
  original_price,
  dlu,
  mysql_dlu
FROM action_sku_dim
WHERE campaign_id = 35023
  AND sku IN ('B08K3LZ4KL', 'B09SFFJP1P')
ORDER BY sku, date_from;


SELECT
  campaign_id,
  sku,
  COUNT(*) AS open_rows
FROM action_sku_dim
WHERE date_to = '2199-12-31 23:59:59'
GROUP BY campaign_id, sku
HAVING COUNT(*) > 1;
