  publisher_id,
  COUNT(*)                        AS clicks,
  COUNT(DISTINCT click_id)        AS unique_click_ids,
  MIN(event_datetime)             AS first_seen,
  MAX(event_datetime)             AS last_seen
FROM irdwm.irdw_8900_squarespace_click_id_investigation
GROUP BY publisher_id
ORDER BY clicks DESC;
