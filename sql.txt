FROM radius_path_event_fact r
WHERE r.campaign_id = 9488
  AND r.conversion_event_id IN (
    'A-9488.6475.13560',
    'A-9488.6476.1182803',
    'A-9488.6480.169991',
    'A-9488.6480.717713'
  )
  AND r.event_type = 'Click';

SELECT cf.publisher_id, COUNT(*) AS clicks,
       MIN(cf.event_datetime) AS first_seen, MAX(cf.event_datetime) AS last_seen
FROM click_fact cf
WHERE cf.campaign_id = 9488
  AND cf.landing_page_url LIKE 'https://thefarmersdog.com/p/rakutendpd%'
  AND cf.event_datetime BETWEEN '2025-09-22' AND '2025-09-30'
GROUP BY cf.publisher_id
ORDER BY clicks DESC;
