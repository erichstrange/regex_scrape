-- Auto-reversed (and other reversed) actions for OurRitual since Sep 1, 2025
SELECT
  action_id,
  oid,
  campaign_id,
  publisher_id,
  adj_reason,
  new_status,
  adj_date,
  batch_id
FROM irac_action2
WHERE campaign_id = 24744
  AND new_status = 'REVERSED'
  AND adj_date >= '2025-09-01'
ORDER BY adj_date DESC;
