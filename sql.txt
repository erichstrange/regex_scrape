-- CONVERSIONS API LOGS FOR AXS (ir_api_log)
SELECT *
FROM ir_api_log
WHERE iram_account_id = 4984840
  AND doe >= '2025-11-11' AND doe < '2025-11-12'
ORDER BY submit_date DESC;


-- AUDIT: TechnicalIntegration / Gateway settings for account
SELECT *
FROM m_audit_log
WHERE account_id = 4984840
  AND entity_type = 'TechnicalIntegration'
ORDER BY change_date DESC;
