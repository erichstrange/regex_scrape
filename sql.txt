-- Manual reversals since Sep 1, 2025
SELECT
  action_id,
  oid,
  campaign_id,
  publisher_id,
  adj_reason,
  new_status,
  adj_date,
  batch_id
FROM irac_action2
WHERE campaign_id = 24744             -- OurRitual campaign
  AND new_status = 'REVERSED'
  AND adj_date >= '2025-09-01'
ORDER BY adj_date DESC;


-- DRAFT: OurRitual actions with Event Risk indicators or reversed since Sep 1, 2025
-- [DataSource] datalake_irdwm.action_fact
SELECT
  action_datetime,
  action_id,
  oid,
  advertiser_id,
  campaign_id,
  publisher_id,
  approval_status,        -- includes REVERSED / APPROVED / etc.
  adj_reason,             -- manual modification note, if any
  fraud_spoofing          -- Event Risk flag for conversion spoofing
FROM `datalake_irdwm.action_fact`
WHERE advertiser_id = 5399434        -- OurRitual account
  AND campaign_id   = 24744
  AND action_datetime >= TIMESTAMP('2025-09-01')
  AND (approval_status = 'REVERSED' OR fraud_spoofing = TRUE)
ORDER BY action_datetime DESC;
