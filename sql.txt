-- Tracker definition
SELECT * FROM ircm_actiontracker WHERE id = 60393;

-- If this is a mobile app tracker, check mobile app-specific config
SELECT
  at.ircm_campaign_id, c.name AS campaign_name,
  at.name AS tracker_name, m.platform, m.download_url
FROM ircm_mobileapp_actiontracker m
JOIN ircm_actiontracker at ON m.id = at.id
JOIN ircm_campaign c ON c.id = at.ircm_campaign_id
WHERE at.id = 60393;


SELECT * FROM ircm_technicalintegration WHERE ircm_campaign_id = 19183;


-- Signed contracts (active = current IOs)
SELECT * FROM ircm_signedcontract WHERE ircm_campaign_id = 19183;

-- If you need credit group / IO term detail
SELECT sc.iram_publisher_id, sc.ircm_campaign_id, iot.credit_group
FROM ircm_signedcontract sc
LEFT JOIN ircm_ioterms iot ON iot.id = sc.ircm_ioterms_id
WHERE sc.ircm_campaign_id = 19183
  AND sc.date_contract_end IS NULL;
