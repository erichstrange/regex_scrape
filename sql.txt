-- Was this address even in the generated audience?
SELECT id, outreach_id, to_email, subject, created_at, status, sent_at, provider, provider_message_id
FROM new_email
WHERE outreach_id IN (74055, 74060)
  AND lower(to_email) = 'info@mimoni.com'
ORDER BY created_at DESC;

-- Any evidence we ever handed anything off to SG?
SELECT COUNT(*) AS with_provider_ids
FROM new_email
WHERE outreach_id IN (74055, 74060)
  AND provider = 'sendgrid'
  AND provider_message_id IS NOT NULL;
