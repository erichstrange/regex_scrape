-- CURRENT RULES ON THE TRACKER (Active trackers + patterns)
SELECT  C.iram_advertiser_id,
        M.ircm_campaign_id,
        M.id              AS tracker_id,
        M.name            AS tracker_name,
        M.state,
        M.trackerType,
        M.method,
        M.activation_date,
        M.deactivation_date,
        M.oid_pattern,           -- Order ID regex (allow-list)
        M.url_pattern,           -- Conversion page URL regex (allow-list)
        M.dlu, M.ulu, M.doe, M.uoe
FROM ircm_actiontracker M
LEFT JOIN ircm_campaign C ON M.ircm_campaign_id = C.id
WHERE M.ircm_campaign_id = 25209
  AND M.id = 46312;


SELECT change_date, username, auditedvariable, previousval, newval
FROM m_audit_log_entry
WHERE objectclassname = 'estalea.ir.cm.campaign.domain.TechnicalIntegration'
  AND objectid = 25209
  AND newval != previousval
ORDER BY change_date DESC
LIMIT 400;


SELECT *
FROM ircm_action_credit_rule_history
WHERE ircm_actiontracker_id = 46312
ORDER BY date_started DESC;
