SELECT ne.outreach_id, ne.email_address, ne.subject, ne.created_at
FROM outreach.new_email ne
WHERE ne.created_at BETWEEN '2025-09-23 03:30:00' AND '2025-09-23 04:45:00'
ORDER BY ne.created_at DESC
LIMIT 1000;


SELECT j.*
FROM outreach.icee_job j
JOIN outreach.new_email ne ON ne.outreach_id = j.outreach_id
WHERE ne.email_address = 'marketing@egofitwalker.com'
  AND ne.created_at BETWEEN '2025-09-22 00:00:00' AND '2025-09-24 23:59:59'
ORDER BY j.id DESC
LIMIT 200;
