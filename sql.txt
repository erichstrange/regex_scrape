SELECT IA.id,
       IA.doe,
       P.platform_approval_state
FROM   iram_account          IA
LEFT   JOIN ircm.ircm_pendingcontract PC ON PC.iram_publisher_id = IA.id
LEFT   JOIN ircm.ircm_signedcontract  SC ON SC.iram_publisher_id  = IA.id
JOIN   iram.iram_publisher    P  ON   P.id = IA.id
WHERE  IA.id = 6300735
  AND  PC.iram_publisher_id IS NULL
  AND  SC.iram_publisher_id IS NULL;
