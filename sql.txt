SELECT
  ne.outreach_id,
  ne.outreach_type,
  ne.message_id,
  ne.email_address,
  ne.subject,
  ne.created_at
FROM outreach.new_email ne
WHERE ne.email_address = 'marketing@egofitwalker.com'
  AND ne.created_at BETWEEN '2025-09-22 00:00:00' AND '2025-09-24 23:59:59'
ORDER BY ne.created_at DESC
LIMIT 200;

-- DS: r_ds_iram
SELECT a.feature_key AS 'Key', a.name AS 'Feature Name'
FROM iram_feature a
JOIN iram_feature_auth_account_feature b ON a.id = b.iram_feature_id
JOIN iram_feature_auth_account c ON c.id = b.iram_feature_auth_account_id
WHERE c.id IN (
  SELECT iram_feature_auth_account_id
  FROM iram_account d
  WHERE d.id = 6488158
)
ORDER BY a.feature_key;
