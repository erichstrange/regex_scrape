       id as campaign_id,
       name, state, platform_approval_state,
       setup_complete_date, request_activation_date,
       activation_date, deactivation_date, doe, dlu, uoe, ulu
from ircm_campaign
where name like '%Hanna Andersson%'
limit 1000;

SELECT sc.iram_publisher_id, sc.ircm_campaign_id, iot.credit_group
FROM ircm_signedcontract sc
LEFT JOIN ircm_ioterms iot ON iot.id = sc.ircm_ioterms_id
WHERE sc.iram_publisher_id = '12869'
  AND sc.ircm_campaign_id = <HannaCampaignId>
  AND sc.date_contract_end IS NULL;

from ircm_pendingcontract pen

       auto_corrected_partner_id      AS partner_we_assigned,
       status,
       type                           AS problem_type,

INNER JOIN icomp.search_content_aggregator_log AS `log`
        ON `log`.compliance_batch_item_id = content_item_link.compliance_batch_item_id
