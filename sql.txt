SELECT
  id,
  state,
  debit_amount,
  debit_amount_currency,
  due_date,
  charge_date,
  link_date,
  last_notified_date,
  description,
  reference,
  parent_id,
  doe,
  dlu
FROM finance_invoice
WHERE seller_finance_account_id = 2198017
  AND reference = '215785'
ORDER BY id;


SELECT
  id,
  state,
  debit_amount,
  debit_amount_currency,
  credit_amount,
  credit_amount_currency,
  due_date,
  charge_date,
  description,
  reference,
  seller_finance_account_id,
  buyer_finance_account_id,
  doe,
  dlu
FROM finance_invoice
WHERE
(
  seller_finance_account_id = 2198017
  OR buyer_finance_account_id = 2198017
)
AND charge_date >= '2025-12-08'
AND charge_date <  '2025-12-18'
ORDER BY charge_date DESC, id DESC;
