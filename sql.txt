SELECT
       d.action_id,
       d.message                AS reversal_reason,   -- use d.code if thatâ€™s clearer
       d.doe                    AS reversed_at_utc,
       a.<IP_COLUMN>            AS ip,                -- replace with ref_ip_address OR device_ip
       a.publisher_id,
       a.action_date
FROM   irac_disposition_log AS d
JOIN   irac_action2          AS a
         ON a.id = d.action_id
WHERE  a.iram_advertiser_id  = 5333971          -- Brooks Running
  AND  d.doe >= DATE_SUB(NOW(), INTERVAL 30 DAY)
ORDER  BY d.doe DESC
LIMIT 500;
