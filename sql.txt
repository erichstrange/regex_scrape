
SELECT 
  cf.event_datetime,
  cf.advertiser_id,
  cf.campaign_id,
  cf.publisher_id,
  cf.landing_page_url,
  cf.referring_url,
  cf.classification,
  cf.method
FROM click_fact cf
WHERE cf.advertiser_id = 404719              -- Hanna Andersson adv id
  AND cf.campaign_id  = 5644
  AND cf.publisher_id = 12869
  AND cf.event_datetime >= '2025-10-20'
  AND cf.event_datetime <  '2025-10-29'
  AND cf.landing_page_url LIKE '%discountsclubs.com%'
ORDER BY cf.event_datetime DESC;
