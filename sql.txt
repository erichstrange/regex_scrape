-- Finance email log (full history, newest first)
SELECT *
FROM finance_email_log
WHERE iram_account_id = 327676
ORDER BY doe DESC;



-- Audit log entries scoped to InvoiceSettings (verbose entries)
SELECT *
FROM maudit_log_entry
WHERE objectclassname = 'estalea.ir.accounting.invoice.domain.InvoiceSettings'
  AND coreaccountid = 327676
ORDER BY changedate DESC;

-- Companion audit table for InvoiceSettings (summary entries)
SELECT *
FROM maudit_log
WHERE accountid = 327676
  AND entity_type = 'InvoiceSettings'
ORDER BY changedate DESC;



-- Pull invoice headers if you have specific IDs or want to confirm generation
SELECT *
FROM finance_invoice
WHERE iram_account_id = 327676
ORDER BY id DESC;
