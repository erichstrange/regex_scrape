-- [CORE / Auth] What features are enabled for this account?
SELECT a.feature_key AS 'Key', a.name AS 'Feature Name'
FROM iram_feature AS a
JOIN iram_feature_auth_account_feature AS b ON a.id = b.iram_feature_id
JOIN iram_feature_auth_account AS c ON c.id = b.iram_feature_auth_account_id
WHERE iram_feature_auth_account_id IN (
  SELECT iram_feature_auth_account_id
  FROM iram_account AS d
  JOIN core_account AS e ON d.id = e.id
  WHERE d.id = 6488158       -- <-- replace with the advertiser Account ID from the ticket
);


-- [DRAFT] Was an ICEE job created/processed for this outreach?
SELECT *
FROM outreach.icee_job
WHERE outreach_id = :outreach_id
ORDER BY id DESC
LIMIT 200;


-- [DRAFT] What did Outreach log around this send?
SELECT *
FROM outreach.outreach_log
WHERE outreach_id = :outreach_id
ORDER BY id DESC
LIMIT 200;


-- [DRAFT] Who/what was in the target segment for this job?
SELECT *
FROM outreach.segment_preview
WHERE outreach_id = :outreach_id
LIMIT 200;
