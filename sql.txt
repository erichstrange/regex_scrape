-- Datasource: irdw_imp (SingleStore)
SELECT
  oid,
  action_id,
  platform AS device,
  device_family AS device_type,
  os_name AS OS,
  ua_name AS User_Agent
FROM action_fact fact
INNER JOIN client_dim cld
  ON cld.id = client_dim_id
WHERE
  fact.action_datetime BETWEEN '2025-09-29 00:00:00' AND '2025-10-06 23:59:59'
  AND advertiser_id = 5804675
  AND campaign_id  = 33961
  AND oid IN (
    'US-12062826-S','US-12060804-S','US-12043560-S','US-12031873-S','US-12031307-S'
  );

-- Datasource: iraction_sharded
SELECT
  action_id, oid, campaign_id, ircm_terms_id, payout_trace, sku,
  ircm_category_list_id, doe
FROM irac_action2
WHERE campaign_id = 33961
  AND oid IN ('US-12062826-S','US-12060804-S','US-12043560-S','US-12031873-S','US-12031307-S')
ORDER BY doe;
