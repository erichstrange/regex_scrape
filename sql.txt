/* Count partner vs media-source clicks */
SELECT
  CASE WHEN publisher_id = 0 THEN 'MEDIA_SOURCE_ORG/DIR' ELSE 'PARTNER' END AS source_type,
  COUNT(*) AS clicks
FROM click_fact
WHERE advertiser_id = 5439384
  AND campaign_id   = 25140
  AND event_datetime >= '2025-04-25' AND event_datetime < '2025-05-09'
GROUP BY 1;
