-- Base: vetted query from conversion_fact (all methods: JS, API, FTP)
-- DATERANGE: narrow to test window
SELECT event_datetime, method, campaign_id, action_tracker_id,
       oid, currency, amount, items, json
FROM conversion_fact
WHERE oid IN ('80002205368323')
  AND action_tracker_id IN (60393)
  AND event_datetime >= '2025-10-23' AND event_datetime < '2025-10-24'
LIMIT 1000;
