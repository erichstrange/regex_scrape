SELECT
  st.id,
  st.ircm_campaign_id AS campaign_id,
  ins.name AS template_term_name,
  st.name AS scheduled_term_name,
  st.start_date,
  st.end_date,
  st.archived,
  st.payout_json
FROM ircm_scheduled_terms st
JOIN ircm_insertionorder ins ON ins.id = st.ircm_insertionorder_id
WHERE st.id = 16929;  -- from payout_trace


SELECT id, name, start_date, end_date, archived
FROM ircm_scheduled_terms
WHERE ircm_campaign_id = 17328
ORDER BY start_date DESC;
