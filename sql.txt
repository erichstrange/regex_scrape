-- [r_ds_ods] API logs for an account/time window
SELECT id, batch_id, submit_date, completed_date, type, event_code, error_json
FROM ir_api_log
WHERE iram_account_id = 2984864
  AND doe >= '2025-10-01 13:00:00' -- 9am ET
  AND doe <  '2025-10-01 21:00:00' -- 5pm ET
  AND type = 'Conversion'          -
ORDER BY submit_date;
