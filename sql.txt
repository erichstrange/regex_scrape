SELECT DISTINCT 
  ap.iram_publisher_id,
  ca.taxclass,
  ca.organizationname
FROM iram_advertiser_publisher as ap
INNER JOIN iram.core_account as ca
  ON ap.iram_publisher_id=ca.id 
INNER JOIN iram.iram_account as ia
  ON ap.iram_publisher_id=ia.id
WHERE iram_advertiser_id=121062
  AND ap.date_expired IS NULL 
  AND ca.state="ACTIVE"
  AND ca.taxclass IS NOT NULL
  AND NOT ca.taxclass = " "
  AND ia.state= "ACTIVE"
ORDER BY organizationname;
