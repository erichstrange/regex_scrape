SELECT *
FROM irbase_postback_config
WHERE iram_account_id = 3830263;



SELECT C.iram_advertiser_id,
       M.ircm_campaign_id,
       C.name AS "campaign_name",
       M.id AS "tracker_id",
       M.name AS "tracker_name",
       M.codes,
       M.state,
       M.trackerType,
       M.is_parent,
       M.method,
       M.activation_date,
       M.deactivation_date
FROM ircm_actiontracker AS M
LEFT JOIN ircm_campaign AS C
ON M.ircm_campaign_id = C.id
WHERE M.state = 'ACTIVE'
AND M.ircm_campaign_id in (41441);
