/* REI – campaign/partner/time-scoped actions
   Window: 2025-10-14 00:00:00 UTC to 2025-10-17 19:00:00 UTC
*/
SELECT
  action_id,
  oid,
  campaign_id,
  action_tracker_id,
  publisher_id,
  action_date
FROM ir_correlated_event
WHERE campaign_id = 17195
  AND publisher_id IN (5879071,2536211,2850304,6213049,2022587)
  AND action_date >= TIMESTAMP '2025-10-14 00:00:00'
  AND action_date <  TIMESTAMP '2025-10-17 19:00:00'
ORDER BY action_date;


/* Append these filters to the vetted "Action SKU – Product Catalog Match" query:
   - Time window
   - Campaign (via action_id prefix if supported)    <-- see note
   - Category-path patterns from REI’s payout groups
*/
WHERE
  doe >= '2025-10-14 00:00:00' AND doe < '2025-10-17 19:00:00'
  /* Some environments allow filtering campaign via the left-most part of action_id (e.g., '17195.%').
     If supported, uncomment the next line; if not, skip it—I'll reconcile to Query 1 by action_id. */
  -- AND action_id LIKE '17195.%'
  AND (
    original_format_category LIKE 'Sporting Goods > Outdoor Recreation > Winter Sports & Activities > Skiing & Snowboarding%'
    OR original_format_category LIKE 'Sporting Goods > Outdoor Recreation > Winter Sports & Activities > Snowshoeing%'
    OR original_format_category LIKE 'Apparel & Accessories > Clothing > Outerwear > Snow Pants & Suits%'
    OR original_format_category LIKE 'Vehicles & Parts > Vehicle Parts & Accessories > Vehicle Storage & Cargo > Motor Vehicle Carrying Racks > Vehicle Ski & Snowboard Racks%'
    /* Safety net if only the normalized leaf is present on your base query: */
    OR category LIKE 'Skiing & Snowboarding%'
    OR category LIKE 'Snowshoeing%'
    OR category LIKE 'Snow Pants & Suits%'
    OR category LIKE 'Vehicle Ski & Snowboard Racks%'
  )
ORDER BY doe;
