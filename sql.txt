
/* 1. Active ads that carry the label */
SELECT a.id            AS ad_id,
       a.name,
       a.state,
       l.label_name,
       a.last_updated
FROM   ad                 a
JOIN   adlabel_map        m  ON m.ad_id   = a.id
JOIN   adlabel            l  ON l.id      = m.adlabel_id
WHERE  a.account_id       = 2797309
  AND  a.state            = 'ACTIVE'
  AND  l.label_name       = 'Sell more with Shopify'
ORDER  BY ad_id;

/* 2. Which partners the label is indexed to */
SELECT partner_id,
       COUNT(*) AS ads_in_index
FROM   partner_label_index
WHERE  account_id = 2797309
  AND  label_name = 'Sell more with Shopify'
GROUP  BY partner_id
ORDER  BY ads_in_index DESC;
