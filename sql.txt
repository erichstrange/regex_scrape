-- Upcoming scheduled terms + metadata about who created/updated them
SELECT ins.name AS template_term_name, st.name AS scheduled_term_name, st.start_date, st.end_date,
       st.archived, st.doe, st.uoe, st.dlu, st.ulu
FROM ircm_scheduled_terms st
JOIN ircm_insertionorder ins ON ins.id = st.ircm_insertionorder_id
WHERE st.ircm_campaign_id = <campaignId>
ORDER BY st.start_date DESC;
