-- Users & last logins for the advertiser account
SELECT u.id, ce.username, ce.email, ce.firstname, ce.lastname, u.state, ce.lastlogin_date, ce.doe, ce.dlu, ce.uoe, ce.ulu
FROM iram_usership u
JOIN core_estaleauser ce ON u.account_user_id = ce.id
WHERE u.irAccount_Id = "<advertiserId>"
ORDER BY ce.lastlogin_date DESC;


SELECT *
FROM ir_api_log
WHERE iram_account_id = <advertiserId>
  AND doe >= "<YYYY-MM-DD>" AND doe < "<YYYY-MM-DD>";


-- Upcoming scheduled terms + metadata about who created/updated them
SELECT ins.name AS template_term_name, st.name AS scheduled_term_name, st.start_date, st.end_date,
       st.archived, st.doe, st.uoe, st.dlu, st.ulu
FROM ircm_scheduled_terms st
JOIN ircm_insertionorder ins ON ins.id = st.ircm_insertionorder_id
WHERE st.ircm_campaign_id = <campaignId>
ORDER BY st.start_date DESC;
