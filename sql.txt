-- DRAFT – boundary-check for Walmart Creator / oak.haus (4085301)
-- Purpose: show which orders flip in/out of Oct 1–31 when viewed in two time zones.

SELECT
    a.oid,
    a.action_id,
    a.campaign_id,
    a.publisher_id,
    a.doe                          AS doe_utc_raw,
    -- Adjust as needed if your warehouse uses a different function
    (a.doe AT TIME ZONE 'US/Central') AS doe_ct,
    (a.doe AT TIME ZONE 'US/Pacific') AS doe_pt
FROM irac_action2 a
WHERE a.iram_account_id = 1356456               -- Walmart account
  AND a.publisher_id    = 4085301               -- oak.haus.collective
  AND a.doe >= '2025-09-30'
  AND a.doe <  '2025-11-02'
ORDER BY a.doe;
