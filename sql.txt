-- [Singlestore] action_sku_dim by SKU for Adv 30431
SELECT *
FROM action_sku_dim
WHERE campaign_id = 30431          -- program id for this advertiser
  AND sku = 'B0FVXZRVYR'
ORDER BY date_from DESC;


-- [Singlestore] Map actions on this SKU to SKU dimension rows
SELECT
  a.action_id,
  a.oid,
  a.sku,
  a.quantity,
  a.adv_revenue,
  a.payout,
  a.sku_dim_id,
  s.catalog_id,
  s.catalog_item_id,
  s.original_format_category,
  s.category   AS catalog_category,
  s.subcategory
FROM action_item_fact a
JOIN action_sku_dim s
  ON s.sku = a.sku
 AND s.campaign_id = a.campaign_id
WHERE a.campaign_id = 30431
  AND a.sku = 'B0FVXZRVYR'
  AND a.doe BETWEEN '2025-11-11' AND '2025-11-24';
