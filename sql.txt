-- VETTED: Retrieve all SKU actions for a specific campaign and SKU
-- Datasource: singlestore (action_sku_dim)
SELECT *
FROM action_sku_dim
WHERE campaign_id = {campaign_id}       -- Nutricost Amazon Seller campaign id
  AND sku = "{sku]"
ORDER BY date_from DESC;


{campaign_id} = Nutricost Amazon Seller campaign (likely 35023 from the UI).

{sku]} = "B08K3LZ4KL"
