-- Data source: rti.rti_success_login_history
SELECT event_date,
       login_username,
       ip
FROM rti.rti_success_login_history
WHERE account_id = 5811184
ORDER BY event_date DESC
LIMIT 10;

-- Data source: rti.rti_failed_login_history   (same schema)
SELECT event_date,
       login_username,
       fail_reason
FROM rti.rti_failed_login_history
WHERE account_id = 5811184
  AND event_date > NOW() - INTERVAL 3 DAY
ORDER BY event_date DESC;
