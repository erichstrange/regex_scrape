SELECT st.id               scheduled_term_id,
       st.name             scheduled_term_name,
       st.startdate        scheduled_term_startdate,
       st.enddate          scheduled_term_enddate,
       st.state            scheduled_term_state,
       at.id               actiontracker_id,
       at.name             actiontracker_name,
       at.state            actiontracker_state,
       ioterms.iotype      ioterm_iotype,
       ioterms.state       ioterm_state
FROM ircm_scheduled_terms st
JOIN ircm_actiontracker at
  ON at.id = st.ircm_actiontracker_id
JOIN ircm_ioterms ioterms
  ON ioterms.id = st.ircm_ioterms_id
WHERE st.ircm_campaign_id = 5048
ORDER BY st.startdate ASC;
