SELECT doe, submit_date, type, event_code, mediapartner_id, ircm_campaign_id,
       uoe, batch_id, LEFT(payload, 1000) AS payload_sample
FROM ir_api_log
WHERE iram_account_id = 5475947
  AND doe >= '2025-09-01' AND doe < '2025-11-01'
  AND (
       LOWER(type)       LIKE '%prospect%' OR
       LOWER(event_code) LIKE '%prospect%' OR
       LOWER(payload)    LIKE '%prospect%' OR
       LOWER(payload)    LIKE '%proposal%' OR
       LOWER(payload)    LIKE '%recruit%'  OR
       LOWER(payload)    LIKE '%discover%'
  )
ORDER BY doe DESC
LIMIT 1000;
