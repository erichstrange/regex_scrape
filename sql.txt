SELECT
  ins.name AS template_term_name,
  st.name  AS scheduled_term_name,
  st.start_date, st.end_date,
  st.archived, st.doe, st.uoe, st.dlu, st.ulu
FROM ircm_scheduled_terms st
JOIN ircm_insertionorder ins ON ins.id = st.ircm_insertionorder_id
WHERE st.ircm_campaign_id = 27240
ORDER BY st.start_date DESC;


SELECT sc.*, iot.credit_group
FROM ircm_signedcontract sc
LEFT JOIN ircm_ioterms iot ON iot.id = sc.ircm_ioterms_id
WHERE sc.iram_publisher_id = 1418120
  AND sc.ircm_campaign_id = 27240
  AND sc.date_contract_end IS NULL;
