SELECT u.id AS usership_id, upa.product_enabled, upa.campaign_json, upa.dlu, upa.ulu
FROM iram_user_product_access upa
JOIN iram_usership u ON u.id = upa.iram_usership_id
JOIN core_estaleauser ce ON ce.id = u.account_user_id
WHERE u.irAccount_Id = 5399049
  AND ce.email='josh.conyard@exinity.com';

SELECT u.id AS usership_id, u.account_user_id, u.irAccount_Id, u.state,
       ce.id AS estalea_user_id, ce.email, ce.username, ce.firstname, ce.lastname, ce.lastlogin_date
FROM iram_usership u
JOIN core_estaleauser ce ON u.account_user_id = ce.id
WHERE u.irAccount_Id = 5399049
  AND ce.email='josh.conyard@exinity.com'
ORDER BY ce.lastlogin_date DESC;


SELECT * FROM core_role
WHERE id IN (
  SELECT core_role_id
  FROM core_ap_role
  WHERE core_ap_id = (
    SELECT authorizationPrincipal_id
    FROM iram_usership
    WHERE account_user_id = <account_user_id_from_step1>
      AND irAccount_id = 5399049
  )
);


SELECT p.*
FROM core_permission p
JOIN core_role_permission rp ON rp.core_permission_id = p.id
WHERE rp.core_role_id = <role_id_from_step3>;
