SELECT
  C.iram_advertiser_id,
  M.ircm_campaign_id,
  C.name AS campaign_name,
  M.id   AS tracker_id,
  M.name AS tracker_name,
  M.trackerType,
  M.method,
  M.state,
  M.activation_date,
  M.deactivation_date
FROM ircm_actiontracker AS M
LEFT JOIN ircm_campaign AS C
  ON M.ircm_campaign_id = C.id
WHERE C.iram_advertiser_id = 1268155
  AND M.state = 'ACTIVE'
ORDER BY M.ircm_campaign_id, M.id;


select
  iram_advertiser_id,
  id as campaign_id,
  name,
  state,
  platform_approval_state,
  setup_complete_date,
  request_activation_date,
  activation_date,
  deactivation_date
from ircm_campaign
where iram_advertiser_id = 1268155
order by activation_date desc
limit 1000;

SELECT
  id   AS campaign_id,
  name,
  state,
  activation_date,
  deactivation_date
FROM ircm_campaign
WHERE iram_advertiser_id = 1268155
ORDER BY activation_date DESC;
