SELECT
  sc.iram_publisher_id,
  sc.ircm_campaign_id,
  iot.credit_group
FROM ircm_signedcontract sc
LEFT JOIN ircm_ioterms iot ON iot.id = sc.ircm_ioterms_id
WHERE sc.iram_publisher_id = '408233'
  AND sc.ircm_campaign_id = '4273'
  AND sc.date_contract_end IS NULL;


SELECT *
FROM ircm_signedcontract
WHERE ircm_campaign_id = 4273;


SELECT *
FROM ircm_pendingcontract pen
INNER JOIN ircm_campaign ic ON ic.id = pen.ircm_campaign_id
WHERE pen.iram_publisher_id = 408233;


SELECT it.iram_advertiser_id, io.name, it.timezone, it.active_start_date, it.active_end_date, io.ulu
FROM ircm_ioterms AS it
LEFT JOIN ircm_insertionorder AS io
  ON it.iram_advertiser_id = io.iram_advertiser_id
WHERE it.ircm_campaign_id = 4273;
