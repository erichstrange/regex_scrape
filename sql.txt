-- [iraction_sharded] Check category list usage for this SKU
SELECT 
    action_id,
    oid,
    campaign_id,
    ircm_terms_id,
    payout_group_id,
    ircm_category_list_id,
    sku,
    doe
FROM irac_action2
WHERE campaign_id = 30431
  AND sku = 'B09D8S7883'
  AND doe BETWEEN '2025-12-01' AND '2025-12-06'
ORDER BY doe DESC;
