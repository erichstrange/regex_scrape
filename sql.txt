-- Active mobile app events for given campaign
SELECT
    c.id         AS event_type_id,
    CONCAT(b.name,' (',b.id,')') AS campaign,
    DATE(pe.doe) AS date_created,
    pe.name      AS mobile_event,
    c.platform   AS platform,
    a.state      AS event_type_state
FROM ircm_actiontracker a
JOIN ircm_campaign b ON b.id = a.ircm_campaign_id
JOIN ircm_payout_event pe ON pe.ircm_actiontracker_id = a.id
JOIN ircm_event_type c ON c.id = pe.ircm_event_type_id
WHERE b.id = 11703;
