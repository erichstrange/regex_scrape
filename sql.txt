SELECT
    cf.event_datetime,
    cf.advertiser_id,
    cf.campaign_id,
    cf.publisher_id,
    cf.landing_page_url,
    cf.classification,
    cf.method
FROM click_fact cf
WHERE cf.advertiser_id = 4012120          -- Garnet Hill advertiser id
  AND cf.publisher_id = 10078             -- Skimlinks
  -- AND cf.campaign_id = <GARNET_HILL_CAMPAIGN_ID>  -- optional if you know it
  AND cf.event_datetime >= '2025-11-24'   -- start_date
  AND cf.event_datetime <  '2025-12-04'   -- end_date (exclusive)
  AND cf.landing_page_url ILIKE '%bestcodes.org%'
ORDER BY cf.event_datetime DESC;
