(high-level batch status)	sql -- List the 25 most-recent FTP batches (any file) SELECT b.id AS batch_id, bf.file_name, b.status, b.submit_time FROM trackk_ftp.ftp_batch b JOIN trackk_ftp.ftp_batch_file bf ON b.id = bf.batch_id WHERE bf.file_name LIKE 'pvc_%' -- ← narrow to your file(s) ORDER BY b.submit_time DESC LIMIT 25;	Schema is trackk_ftp.
