/* ds_iraction_sharded */

SELECT
       d.action_id,
       d.disposition_code,
       d.disposition_reason,
       d.created_datetime  AS reversed_at_utc,
       a.ref_ip_address,
       a.partner_id,
       a.event_date
FROM   irac_disposition_log AS d
JOIN   irac_action2          AS a
       ON a.id = d.action_id            -- tie log to full action record
WHERE  a.iram_advertiser_id = 5333971   -- Brooks Running
  AND  d.created_datetime   >= DATE_SUB(NOW(), INTERVAL 30 DAY)
ORDER  BY d.created_datetime DESC
LIMIT  500;
